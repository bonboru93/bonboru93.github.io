加载为crash工具编写的功能扩展

扩展的代码格式可以类比proc，首先要有一个command table:

struct command_table_entry {
            char *name;
            cmd_func_t func;
            char **help_data,
            ulong flags;
    };

其中包括命令的名字，实际执行的代码，帮助信息，以及一个标志位，标志位包含REFRESH_TASK_TABLE和MINIMAL两个bit，具体用途还不明确。

下面是一个范例扩展，实现了和echo命令相同的功能：

#include "defs.h"      /* From the crash source top-level directory */



void echo_init(void);    /* constructor function */
void echo_fini(void);    /* destructor function (optional) */

void cmd_echo(void);     /* Declare the commands and their help data. */
char *help_echo[];

static struct command_table_entry command_table[] = {
        { "echo", cmd_echo, help_echo, 0},          /* One or more commands, */
        { NULL },                                     /* terminated by NULL, */
};


void __attribute__((constructor))
echo_init(void) /* Register the command set. */
{ 
        register_extension(command_table);
}
 
/* 
 *  This function is called if the shared object is unloaded. 
 *  If desired, perform any cleanups here. 
 */
void __attribute__((destructor))
echo_fini(void) { }


/* 
 *  Arguments are passed to the command functions in the global args[argcnt]
 *  array.  See getopt(3) for info on dash arguments.  Check out defs.h and
 *  other crash commands for usage of the myriad of utility routines available
 *  to accomplish what your task.
 */
void
cmd_echo(void)
{
        int c;

        while ((c = getopt(argcnt, args, "")) != EOF) {
                switch(c)
                {
                default:
                        argerrs++;
                        break;
                }
        }

        if (argerrs)
                cmd_usage(pc->curcmd, SYNOPSIS);

        while (args[optind]) 
                fprintf(fp, "%s ", args[optind++]);

        fprintf(fp, "\n");
}

/* 
 *  The optional help data is simply an array of strings in a defined format.
 *  For example, the "help echo" command will use the help_echo[] string
 *  array below to create a help page that looks like this:
 * 
 *    NAME
 *      echo - echoes back its arguments
 *
 *    SYNOPSIS
 *      echo arg ...
 *
 *    DESCRIPTION
 *      This command simply echoes back its arguments.
 *
 *    EXAMPLE
 *      Echo back all command arguments:
 *
 *        crash> echo hello, world
 *        hello, world
 *
 */
 
char *help_echo[] = {
        "echo",                        /* command name */
        "echoes back its arguments",   /* short description */
        "arg ...",                     /* argument synopsis, or " " if none */
 
        "  This command simply echoes back its arguments.",
        "\nEXAMPLE",
        "  Echo back all command arguments:\n",
        "    crash> echo hello, world",
        "    hello, world",
        NULL
};

rd

官方wiki：https://crash-utility.github.io/help_pages/rd.html

输出某段内存的内容，可以接受地址或记号，可能需要配合其他工具中找出的地址来使用

接受两个参数，一个地址或记号，另一个长度，示例：

常用选项：

按多少位分割：

-s：如果数值是一个地址，显式的表示出来

-S：如果数值是一个地址，且是由slab分配的，输出slab的名字

-SS：如果数值是一个地址，且是由slab分配的，输出slab的名字和地址

官方wiki：https://crash-utility.github.io/help_pages/task.html

可以输入pid，如果不输入默认就显示core的那个那个task

bt

官方wiki：https://crash-utility.github.io/help_pages/bt.html

输出panic位置的调用栈

常用选项：

-c 选取的cpu：输出panic时所有cpu的调用栈

-l：一次性输出调用栈上所有函数的代码位置

-FF：重要选项，显示调用栈的所有内容，可以据此检查传递给函数的参数以及函数的局部变量

编写一个会panic的范例模块验证效果：

#include <linux/kernel.h>
#include <linux/module.h>


static int layer3(int a)
{
    int ia = 888, ib = 999;
    a = a / (ia - 888);
    printk("crash_test=%d\n", a);
    return ia + ib;
}

static int layer2(int a, int b)
{
    return layer3(111);
}

static int layer1(int a, int b, int c)
{
    int ia = 666, ib = 777;
    layer2(456, 789);
    return ia + ib;
}

static int __init bbu_init(void)
{
    layer1(1, 2, 3);
    return 0;
}
static void __exit bbu_exit(void)
{
}
module_init(bbu_init);
module_exit(bbu_exit);
